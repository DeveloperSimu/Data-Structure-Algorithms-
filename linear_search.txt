void linear_search()
{
	printf("\n Enter the number thst has to be searched:");
	scanf("%d",&num);
	for(i=0; i<n; i++)
		{	printf("\n......compare %d with %d",arr[i],num);
			if(arr[i]==num)
				{
					found=1;
					pos=i;

					break;
				}
		}
	if(found==0)
		printf("\n %d does not exist in the array",num);
}
